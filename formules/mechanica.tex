		    \item $ p = \frac{b^2}{a} = - \frac{mC^2}{k} $
		    \item  $ e = \frac{c}{a} = \sqrt{ 1 + \frac{2mC^2 E_0}{k^2} }$
	    \end{itemize}
    \item Geostationaire oplossing: $ v_0 = \sqrt{ \frac{gR^2}{r_0} }$ 
    \end{enumerate}
